@pytest.fixture(autouse=True)
def cleanup_matplotlib():
    """
    FIXTURE: Clean up matplotlib state between tests

    SCOPE: function

    PURPOSE:
        Matplotlib maintains internal state (figures, styles, etc.).
        This fixture ensures tests don't interfere with each other.

    BEHAVIOR:
        - Runs BEFORE each test (setup)
        - Clears matplotlib figures
        - Resets matplotlib style
        - Runs AFTER each test (cleanup)
        - Closes any open figures

    ⚠️  autouse=True means this runs automatically, no need to request it

    WHY IT MATTERS:
        Without this:
        - Test 1 creates figure
        - Test 2 inherits figure from Test 1
        - Tests interfere with each other (flaky)

        With this:
        - Each test starts fresh
        - No interference

    AI CONFIDENCE: ~95%
        Standard matplotlib testing practice.

    WHAT IT DOES:
        Before test:
        - Clears all figures
        - Resets to default style

        After test:
        - Closes remaining figures
        - Clears figure cache
    """
    # SETUP: Run before test

    # Close any existing figures
    plt.close("all")

    # Reset to default style
    plt.rcdefaults()

    # Yield control to test
    yield

    # CLEANUP: Run after test
    # Close any figures test created
    plt.close("all")

    # Reset style again
    plt.rcdefaults()
